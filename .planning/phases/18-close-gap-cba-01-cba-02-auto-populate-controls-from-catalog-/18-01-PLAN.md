<objective>
Wire `get_controls()` from `catalog.py` into `ChaseBlissDevice.from_raw_yaml()` so that when
a device YAML specifies a recognized CBA model name, controls are auto-populated from the
catalog without requiring users to manually inline the full control list.

Root cause of the gap: `ChaseBlissConfig` has no `model` field, so `from_raw_yaml()` has no
signal to trigger catalog lookup. `get_controls()` was defined in Phase 14 but has zero
production callers — catalog constants for CBA-01 (Wombtone MkII) and CBA-02 (Brothers AM)
are inert unless controls are manually inlined in device YAML.

Fix:
1. Add `model: str | None = None` to `ChaseBlissConfig`
2. In `from_raw_yaml()`, after constructing `config`, if `config.controls` is empty and
   `config.model` is set, call `get_controls("Chase Bliss Audio", config.model)` and replace
   `config.controls` via `model_copy(update={"controls": catalog_controls})`
3. Write 4 targeted tests covering: known model → controls populated, unknown model → empty,
   explicit controls → not overwritten, no model → empty (backward compat)

Manufacturer is hardcoded as "Chase Bliss Audio" because this is the CBA plugin — model is
the only variable discriminator needed.
</objective>

        presets=presets,
    )

# After
@classmethod
def from_raw_yaml(cls, data: dict[str, Any]) -> ChaseBlissDevice:
    from rig_chasebliss.catalog import get_controls
    from rig_chasebliss.preset import DigitalPreset

    config = ChaseBlissConfig(**(data.get("config") or {}))
    if not config.controls and config.model:
        catalog_controls = get_controls("Chase Bliss Audio", config.model)
        if catalog_controls:
            config = config.model_copy(update={"controls": catalog_controls})
    presets = [DigitalPreset(**p) for p in (data.get("presets") or [])]
    return cls(
        id=data["id"],
        name=data.get("name", ""),
        type=DeviceType.DIGITAL,
        config=config,
        presets=presets,
    )
```

Key invariants:
- `if not config.controls` — only auto-populate when controls are absent; explicit YAML controls take precedence
- `if config.model` — no lookup when model is None; `get_controls()` not called unnecessarily
- `if catalog_controls` — unknown model returns `[]` from `get_controls()`; no replacement when empty

# --- catalog auto-population via from_raw_yaml ---

def test_from_raw_yaml_known_model_populates_controls():
    data = {
        "id": "wombtone",
        "name": "Wombtone MkII",
        "config": {"type": "chase_bliss", "midi_channel": 1, "model": "Wombtone MkII"},
        "presets": [],
    }
    device = ChaseBlissDevice.from_raw_yaml(data)
    assert len(device.config.controls) > 0, "controls should be populated from catalog"
    cc_numbers = {c.midi_cc for c in device.config.controls}
    # Wombtone MkII has CC14-21
    for cc in range(14, 22):
        assert cc in cc_numbers, f"CC{cc} missing from auto-populated Wombtone controls"


def test_from_raw_yaml_unknown_model_leaves_controls_empty():
    data = {
        "id": "mystery",
        "name": "Mystery Pedal",
        "config": {"type": "chase_bliss", "midi_channel": 2, "model": "Unknown Pedal"},
        "presets": [],
    }
    device = ChaseBlissDevice.from_raw_yaml(data)
    assert device.config.controls == [], "unknown model should leave controls empty"


def test_from_raw_yaml_explicit_controls_not_overwritten():
    explicit = [{"name": "volume", "type": "knob", "midi_cc": 15}]
    data = {
        "id": "wombtone",
        "name": "Wombtone MkII",
        "config": {
            "type": "chase_bliss",
            "midi_channel": 1,
            "model": "Wombtone MkII",
            "controls": explicit,
        },
        "presets": [],
    }
    device = ChaseBlissDevice.from_raw_yaml(data)
    assert len(device.config.controls) == 1, "explicit controls should not be replaced by catalog"
    assert device.config.controls[0].name == "volume"


def test_from_raw_yaml_no_model_leaves_controls_empty():
    data = {
        "id": "mood",
        "name": "Mood MkII",
        "config": {"type": "chase_bliss", "midi_channel": 3},
        "presets": [],
    }
    device = ChaseBlissDevice.from_raw_yaml(data)
    assert device.config.controls == [], "no model field should leave controls empty"
```
